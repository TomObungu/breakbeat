project(breakbeat)

add_library(glad ${PROJECT_SOURCE_DIR}/src/glad.c)
target_include_directories(glad PUBLIC C:/Users/deeza/OneDrive/breakbeat/include)

add_library(SDL2main STATIC IMPORTED)
set_target_properties(SDL2main PROPERTIES IMPORTED_LOCATION ${PROJECT_SOURCE_DIR}/lib/SDL2main.lib)

set_target_properties(SDL PROPERTIES IMPORTED_LOCATION ${PROJECT_SOURCE_DIR}/lib/SDL2.lib)

    "${SOURCE_DIR}/*.c"

    "${SOURCE_DIR}/*.hpp"
    "${SOURCE_DIR}/*.cpp"
    "include/*.hpp"
    "include/*.cpp"
    "include/*.h"
    "include/*.c"
